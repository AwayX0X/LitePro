function Read-Host-Animated {
    param([string]$PromptText, [string]$TitleText)
    
    $FullText = (($([char]0x20) + $([char]0x20) + $([char]0x20) + $([char]0x20)) + $TitleText + ([System.Text.Encoding]::UTF8.GetString([byte[]](0x20,0x20,0x20,0x20,0x20,0x20,0x20,0x20))))
    Write-Host $PromptText -NoNewline -ForegroundColor Yellow
    
    # วิ่งวนข้อความบน Title Bar รอระหว่างพิมพ์คำสั่ง
    $InputString = ''
    while ($true) {
        if ([Console]::KeyAvailable) {
            $key = [Console]::ReadKey($true)
            if ($key.Key -eq [ConsoleKey]::Enter) {
                Write-Host ''
                break
            }
            if ($key.Key -eq [ConsoleKey]::Backspace) {
                if ($InputString.Length -gt 0) {
                    $InputString = $InputString.Substring(0, $InputString.Length - 1)
                    Write-Host ([System.Text.Encoding]::UTF8.GetString([byte[]](0x60,0x62,0x20,0x60,0x62))) -NoNewline
                }
            } else {
                $InputString += $key.KeyChar
                Write-Host $key.KeyChar -NoNewline -ForegroundColor White
            }
        }
        
        # เลื่อนตัวอักษรไปข้างหน้า
        $host.UI.RawUI.WindowTitle = $FullText
        $FullText = $FullText.Substring(1) + $FullText.Substring(0, 1)
        Start-Sleep -Milliseconds 120
    }
    return $InputString
}
